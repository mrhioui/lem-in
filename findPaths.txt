package lemin

import (
	"Lemin-Project/tools"
	"Lemin-Project/variables"
)

func FindAllpaths(current *variables.Room, end *variables.Room, path []*variables.Room, paths *[][]*variables.Room) {
	current.Visited = true
	path = append(path, current)

	if tools.AreRoomsEqual(current, end) {
		CpPath := path
		*paths = append(*paths, CpPath)
	} else {
		for _, v := range current.Relations {
			if !v.Visited {
				FindAllpaths(v, end, path, paths)
			}
		}
	}
	current.Visited = false
	path = path[:len(path)-1]
}
